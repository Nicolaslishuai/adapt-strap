Getting Started:

Install adapt-strap

  1. Install the library using following command in your terminal: bower install adapt-strap --save
  2. Include the library files in your index.html file: 
    <!-- dependencies -->
<script src="bower_components/jquery/dist/jquery.js"></script>
<script src="bower_components/angular/angular.js"></script>
<script src="bower_components/angular-sanitize/angular-sanitize.js"></script>
<script src="bower_components/bootstrap/js/bootstrap.js"></script>
<link rel="stylesheet" href="bower_components/bootstrap/css/bootstrap.css"/>

<!-- adapt-strap -->
<script src="bower_components/adapt-strap/dist/adapt-strap.min.js"></script>
<script src="bower_components/adapt-strap/dist/adapt-strap.tpl.min.js"></script>
<link rel="stylesheet" href="bower_components/adapt-strap/dist/adapt-strap.min.css"/>
  3. Add adaptv.adaptStrap and ngSenitize modules as a dependency to you main app module: angular.module('myApp', [ 'ngSanitize', // adapt-strap requires ngSanitize 'adaptv.adaptStrap' ]);

Configure Pagination

# ==== Important === adapt-strap needs to know how your backend handles pagination ====:
### Here is the default configuration for the pagination:
$adConfigProvider.paging = {
    request: {
        start: 'skip',
        pageSize: 'limit',
        page: 'page',
        sortField: 'sort',
        sortDirection: 'sort_dir',
        sortAscValue: 'asc',
        sortDescValue: 'desc'
    },
    response: {
        itemsLocation: 'data',
        totalItems: 'pagination.totalCount'
    },
    pageSize: 10,
    pageSizes: [10, 25, 50]
};

### You can globally configure the pagination configuration like this during the app initialization:
angular.module('myApp').config(['$adConfigProvider', function ($adConfigProvider) {
    $adConfigProvider.paging.response = {
        totalItems: 'results.opensearch:totalResults',
        itemsLocation: 'results.data'
    };
}])

### You can override global configuration by providing the paginationConfig object under ajaxConfig object at component usage level

Drag and drop

- simple directives to enable drag and drop functionalities

Code

  • draggable.view.html
  • draggable.ctrl.js 
        <!-- ========== Usage Drag ========== -->
        <li class="list-group-item"
            ad-drag="true"
            ad-drag-data="car"
            ad-drag-begin="onDragStart($data, $dragElement, $event);"
            ad-drag-end="onDragEnd($data, $dragElement, $lastDropElement, $event);"
            ng-repeat="car in models.cars">
          <span>
            <span class="glyphicon glyphicon-th"></span>
            {{ car.name }} - {{ car.modelYear }} ( {{ car.price | currency }} )
          </span>
        </li>
      </ul>

      <!-- ========== Usage Drop ========== -->
      <ul id="my-basket"
          class="list-group"
          ad-drop="true"
          ad-drop-over="onDragOver($data, $dragElement, $dropElement, $event);"
          ad-drop-leave="onDragLeave($data, $dragElement, $dropElement, $event)"
          ad-drop-end="onDrop($data, $dragElement, $dropElement, $event);">
        <li class="list-group-item disabled">My Basket</li>
        <li class="list-group-item"
            ad-drag="true"
            ng-repeat="car in models.basket">
          <span>
            <span class="glyphicon glyphicon-thumbs-up"></span>
            {{ car.name }} - {{ car.modelYear }} ( {{ car.price | currency }} )
          </span>
        </li>
        <li class="list-group-item" ng-hide="models.basket.length">
            <span> Drop here</span>
        </li>
      </ul>

      <ul class="list-group">
        <!-- =========== Drag with handles ============= -->
        <li class="list-group-item"
            ad-drag="true"
            ad-drag-handle="true"
            id="drag-handle-demo">
          <span class="ad-drag-handle glyphicon glyphicon-align-justify"></span>
          Drag me using my handle
        </li>
        <!-- =========== Drag on the whole element ============= -->
        <li class="list-group-item"
            ad-drag="true"
            id="drag-element-demo">
          Drag me without handle
        </li>
        <!-- =========== Prevent drag on parts of the element ============= -->
        <li class="list-group-item"
            ad-drag="true"
            id="drag-prevent-demo">
          Drag me, except <strong ad-prevent-drag>this bold text</strong></li>
        </li>
      </ul>

                {{ selectedArtists }}
            </span>

                                  single-selection-mode="true"
                                  on-item-click="artistClicked"

                                  btn-classes="btn btn-primary btn-xs"
                                  ajax-config="artistsAjaxConfig">
            </ad-infinite-dropdown>
            <p>Selected Items:</p>

            </span>

            <!-- ========== Usage Local Data Source ========== -->
